  <script>

    function calculatePrice() {

      const width = parseFloat(document.getElementById('width').value);

      const height = parseFloat(document.getElementById('height').value);

      const basePrice = parseFloat(document.getElementById('basePrice').value);

      const A1Area = 594 * 841;


      if (width && height && basePrice) {

        const area = width * height;

        const price = basePrice * (area / A1Area);

        document.getElementById('result').textContent = `Total: €${price.toFixed(2)}`;

      } else {

        document.getElementById('result').textContent = 'Please fill in all fields';

      }

    }

  </script>
